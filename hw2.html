public static String encrypt(String s, int k) { char[] arr = s.toCharArray(); int ascii; for(int i = 0; i < arr.length; i++) { ascii = (int) arr[i]; ascii = ascii + k; // System.out.println(ascii); if (ascii > 126) { ascii = 32 + (126 % ascii); // System.out.print("Changed: " + ascii); } arr[i] = (char) ascii; } return String.valueOf(arr); }
